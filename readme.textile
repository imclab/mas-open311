h2. Adding custom ReST-Routes to your @conf/routes.php@ file

pre. /**
 * Routing for Open311 Plugin
 */
Router::connect("/open311/requests",array('plugin' => 'open311' , 'controller' => 'open311','action' => 'requests', '[method]' => 'GET'));

pre. Router::connect("/open311/services",array('plugin' => 'open311' , 'controller' => 'open311','action' => 'services', '[method]' => 'GET'));

pre. Router::connect("/open311/requests/:id",array('plugin' => 'open311' , 'controller' => 'open311','action' => 'request', '[method]' => 'GET'),array('id' => '[0-9]+|[A-Fa-f0-9]{8}-[A-Fa-f0-9]{4}-[A-Fa-f0-9]{4}-[A-Fa-f0-9]{4}-[A-Fa-f0-9]{12}', 'pass' => array('id')));
